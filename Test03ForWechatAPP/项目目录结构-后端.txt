后端API服务目录结构：

backend/
├── app.js                # 主入口
├── routes/               # 路由（API接口）
│   ├── auth.js           # 登录与认证
│   ├── warehouse.js      # 仓库相关接口
│   ├── product.js        # 商品相关接口
│   ├── inbound.js        # 入库相关接口
│   ├── outbound.js       # 出库相关接口
│   ├── order.js          # 订单相关接口
│   ├── flow.js           # 流水相关接口
├── controllers/          # 业务逻辑
├── models/               # 数据模型
├── middlewares/          # 中间件（如权限校验、日志等）
├── utils/                # 工具函数
├── config/               # 配置文件
├── logs/                 # 日志
├── public/               # 静态资源
